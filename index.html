    <div id="body-container">
        <div class="glass-card">
            <h1 id="main-text">Will you be my bestfriend? ❤️</h1>
            <div class="primary-buttons">
                <button id="yesBtn" onclick="sayYes()">Yes 💚</button>
                <div id="no-space">
                     <button id="noBtn" onmouseover="moveNo()" onclick="moveNo()">No ❤️</button>
                </div>
            </div>
            <button id="maybeBtn" onclick="sayMaybe()">Maybe 💛</button>
        </div>
    </div>

    <script>
        /* JS YAHAN HAI */
        let noAttempts = 0;
        let maybeAttempts = 0;
        let noSize = 1;

        function moveNo() {
            noAttempts++;
            const noBtn = document.getElementById('noBtn');
            const yesBtn = document.getElementById('yesBtn');

            if (noAttempts < 15) {
                noBtn.style.position = "fixed";
                const x = Math.random() * (window.innerWidth - 110);
                const y = Math.random() * (window.innerHeight - 50);
                noBtn.style.left = x + 'px';
                noBtn.style.top = y + 'px';
                noSize -= 0.065;
                noBtn.style.transform = `scale(${noSize})`;
            }
            else if (noAttempts === 15) {
                const yesRect = yesBtn.getBoundingClientRect();
                noBtn.style.position = "fixed";
                noBtn.style.left = (yesRect.left + (yesRect.width / 2) - (noBtn.offsetWidth / 2)) + 'px';
                noBtn.style.top = (yesRect.top + (yesRect.height / 2) - (noBtn.offsetHeight / 2)) + 'px';
                noBtn.style.transform = `scale(${noSize})`;
                noBtn.style.pointerEvents = "none";
                noBtn.style.opacity = "0.7";
            }
        }

        function sayMaybe() {
            maybeAttempts++;
            const messages = ["Kyuu maybe? Acha nahi hu kya? 🥺", "Kyu mujhe friend nahi banana chahte? 😭", "Maan jao na yaawrr... itna bhav mat khao!", "Itna natak mat kro, seedhe Yes bolo! 😤", "Nahi maan rahi na? Mat mano! Jao!"];
            if (maybeAttempts <= 5) alert(messages[maybeAttempts - 1]);
            if (maybeAttempts === 5) {
                document.getElementById('maybeBtn').classList.add('fade-out');
                setTimeout(() => { document.getElementById('maybeBtn').style.display = 'none'; }, 1000);
            }
        }

        function sayYes() {
            document.getElementById('main-text').innerHTML = "Congratulations! 🥳🎉<br>Hum dono best friends ban gye! ❤️🌸";
            document.getElementById('yesBtn').style.display = 'none';
            document.getElementById('noBtn').style.display = 'none';
            document.getElementById('maybeBtn').style.display = 'none';
        }
    </script>
